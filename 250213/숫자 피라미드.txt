#include <stdio.h>

int main()
{
    int n;
    scanf("%d", &n);
    
    for (int i = 0; i <= n; i++){
        for (int j = i; j <= n - 1; j++){
            printf(" ");
        }
            for (int y = 1; y <= i; y++){
                printf("%d ", y);
            }
        printf("\n");
    }
}
